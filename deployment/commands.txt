[Unit]
Description=Gunicorn instance
After=network.target
[Service]
User=ubuntu
Group=www-data
WorkingDirectory=/home/ubuntu/Artist_Paintings
Environment="PATH=/home/ubuntu/Artist_Paintings/venv/bin"
ExecStart=/home/ubuntu/Artist_Paintings/venv/bin/gunicorn --workers 3 --bind unix:flask_app.sock -m 007 wsgi:application
[Install]
WantedBy=multi-user.target


server {
    listen 80;
    server_name 18.118.218.82;
    location / {
        include proxy_params;
        proxy_pass http://unix:/home/ubuntu/Artist_Paintings/flask_app.sock;
    }
}


sudo ln -s /etc/nginx/sites-available/flask_app /etc/nginx/sites-enabled
